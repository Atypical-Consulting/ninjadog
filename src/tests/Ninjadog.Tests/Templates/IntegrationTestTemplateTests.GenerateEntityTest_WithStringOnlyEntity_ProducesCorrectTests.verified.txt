    [Fact]
    public async Task FullCrudLifecycle_Category_WorksEndToEnd()
    {
        // Create
        var createRequest = new CreateCategoryRequest
        {
            Name = "Test Name",
            Color = "Test Color"
        };
        var createResponse = await Client.PostAsJsonAsync(BaseUrl, createRequest);
        createResponse.StatusCode.Should().Be(HttpStatusCode.Created);
        var created = await DeserializeAsync<CategoryResponse>(createResponse);

        // Read
        var getResponse = await Client.GetAsync($"{BaseUrl}/{created!.Id}");
        getResponse.StatusCode.Should().Be(HttpStatusCode.OK);

        // Update
        var updateRequest = new UpdateCategoryRequest
        {
            Name = "Test Name Updated",
            Color = "Test Color Updated"
        };
        var updateResponse = await Client.PutAsJsonAsync(
            $"{BaseUrl}/{created.Id}", updateRequest);
        updateResponse.StatusCode.Should().Be(HttpStatusCode.OK);

        // Delete
        var deleteResponse = await Client.DeleteAsync($"{BaseUrl}/{created.Id}");
        deleteResponse.StatusCode.Should().Be(HttpStatusCode.NoContent);

        // Verify deleted
        var verifyResponse = await Client.GetAsync($"{BaseUrl}/{created.Id}");
        verifyResponse.StatusCode.Should().Be(HttpStatusCode.NotFound);
    }
