using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;
using Microsoft.AspNetCore.Authorization;

namespace TestApp.Api.Endpoints;

public partial class CreateTodoItemEndpoint
    : Endpoint<CreateTodoItemRequest, TodoItemResponse>
{
    public ITodoItemService TodoItemService { get; private set; } = null!;

    public override void Configure()
    {
        Post("/todo-items");
        AllowAnonymous();
    }

    public override async Task HandleAsync(CreateTodoItemRequest req, CancellationToken ct)
    {
        var todoItem = req.ToTodoItem();

        await TodoItemService.CreateAsync(todoItem);

        var todoItemResponse = todoItem.ToTodoItemResponse();
        await SendCreatedAtAsync<GetTodoItemEndpoint>(
            new { Id = todoItem.Id }, todoItemResponse, generateAbsoluteUrl: true, cancellation: ct);
    }
}
