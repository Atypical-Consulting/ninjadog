using TestApp.Api.Contracts.Requests;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class DeleteOrderEndpoint(IOrderService orderService)
    : Endpoint<DeleteOrderRequest>
{
    public override void Configure()
    {
        Delete("/orders/{id:int}");
        AllowAnonymous();
    }

    public override async Task HandleAsync(DeleteOrderRequest req, CancellationToken ct)
    {
        var deleted = await orderService.DeleteAsync(req.OrderId);
        if (!deleted)
        {
            await SendNotFoundAsync(ct);
            return;
        }

        await SendNoContentAsync(ct);
    }
}
