using System.Net;

using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using FluentAssertions;

namespace TestApp.IntegrationTests;

/// <summary>
/// Integration tests for the Order CRUD endpoints.
/// Tests the full lifecycle: Create → GetAll → Get → Update → Delete.
/// </summary>
public class OrderEndpointTests : IntegrationTestBase
{
    private const string BaseUrl = "/orders";

    public OrderEndpointTests(CustomWebApplicationFactory factory)
        : base(factory)
    {
    }

    [Fact]
    public async Task CreateOrder_WithValidRequest_ReturnsCreated()
    {
        // Arrange
        var request = new CreateOrderRequest
        {
            CustomerName = "Test CustomerName",
            Total = 19.99m
        };

        // Act
        var response = await Client.PostAsJsonAsync(BaseUrl, request);

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.Created);
        var created = await DeserializeAsync<OrderResponse>(response);
        created.Should().NotBeNull();
        created!.CustomerName.Should().Be("Test CustomerName");
        created!.Total.Should().Be(19.99m);

    }

    [Fact]
    public async Task GetAllOrders_AfterCreating_ReturnsCollection()
    {
        // Arrange
        var request = new CreateOrderRequest
        {
            CustomerName = "Test CustomerName",
            Total = 19.99m
        };
        await Client.PostAsJsonAsync(BaseUrl, request);

        // Act
        var response = await Client.GetAsync(BaseUrl);

        // Assert

        result.Should().NotBeNull();
        result!.Orders.Should().NotBeEmpty();
    }

    [Fact]

        var createResponse = await Client.PostAsJsonAsync(BaseUrl, request);

    }

    [Fact]

    }

    [Fact]
    public async Task UpdateOrder_WithValidRequest_ReturnsOk()
    {
        // Arrange
        var createRequest = new CreateOrderRequest
        {
            CustomerName = "Test CustomerName",
            Total = 19.99m
        };
        var createResponse = await Client.PostAsJsonAsync(BaseUrl, createRequest);
        var created = await DeserializeAsync<OrderResponse>(createResponse);

        var updateRequest = new UpdateOrderRequest
        {
            CustomerName = "Test CustomerName Updated",
            Total = 99.99m
        };

        // Act
        var response = await Client.PutAsJsonAsync(
            $"{BaseUrl}/{created!.OrderId}", updateRequest);

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.OK);
        var updated = await DeserializeAsync<OrderResponse>(response);
        updated.Should().NotBeNull();
        updated!.CustomerName.Should().Be("Test CustomerName Updated");
        updated!.Total.Should().Be(99.99m);

    }

    [Fact]
    public async Task DeleteOrder_WithValidId_ReturnsNoContent()
    {
        // Arrange
        var request = new CreateOrderRequest

            CustomerName = "Test CustomerName",

        var createResponse = await Client.PostAsJsonAsync(BaseUrl, request);
        var created = await DeserializeAsync<OrderResponse>(createResponse);

        // Act
        var response = await Client.DeleteAsync($"{BaseUrl}/{created!.OrderId}");

    [Fact]
    public async Task DeleteOrder_WithInvalidId_ReturnsNotFound()

        var response = await Client.DeleteAsync($"{BaseUrl}/999999");

    [Fact]
    public async Task FullCrudLifecycle_Order_WorksEndToEnd()
    {
        // Create
        var createRequest = new CreateOrderRequest
        {
            CustomerName = "Test CustomerName",
            Total = 19.99m
        };
        var createResponse = await Client.PostAsJsonAsync(BaseUrl, createRequest);
        createResponse.StatusCode.Should().Be(HttpStatusCode.Created);
        var created = await DeserializeAsync<OrderResponse>(createResponse);

        // Read
        var getResponse = await Client.GetAsync($"{BaseUrl}/{created!.OrderId}");
        getResponse.StatusCode.Should().Be(HttpStatusCode.OK);

        // Update
        var updateRequest = new UpdateOrderRequest
        {
            CustomerName = "Test CustomerName Updated",
            Total = 99.99m
        };
        var updateResponse = await Client.PutAsJsonAsync(
            $"{BaseUrl}/{created.OrderId}", updateRequest);
        updateResponse.StatusCode.Should().Be(HttpStatusCode.OK);

        // Delete
        var deleteResponse = await Client.DeleteAsync($"{BaseUrl}/{created.OrderId}");
        deleteResponse.StatusCode.Should().Be(HttpStatusCode.NoContent);

        // Verify deleted
        var verifyResponse = await Client.GetAsync($"{BaseUrl}/{created.OrderId}");
        verifyResponse.StatusCode.Should().Be(HttpStatusCode.NotFound);
    }
