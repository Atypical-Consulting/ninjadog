using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class CreateOrderEndpoint(IOrderService orderService)
    : Endpoint<CreateOrderRequest, OrderResponse>
{
    public override void Configure()
    {
        Post("/orders");
        AllowAnonymous();
    }

    public override async Task HandleAsync(CreateOrderRequest req, CancellationToken ct)
    {
        var order = req.ToOrder();

        await orderService.CreateAsync(order);

        var orderResponse = order.ToOrderResponse();
        await SendCreatedAtAsync<GetOrderEndpoint>(
            new { Id = order.OrderId }, orderResponse, generateAbsoluteUrl: true, cancellation: ct);
    }
}
