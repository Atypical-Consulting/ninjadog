using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class UpdateOrderEndpoint(IOrderService orderService)
    : Endpoint<UpdateOrderRequest, OrderResponse>
{
    public override void Configure()
    {
        Put("/orders/{id:int}");
        AllowAnonymous();
    }

    public override async Task HandleAsync(UpdateOrderRequest req, CancellationToken ct)
    {
        var existingOrder = await orderService.GetAsync(req.OrderId);

        if (existingOrder is null)
        {
            await SendNotFoundAsync(ct);
            return;
        }

        var order = req.ToOrder();
        await orderService.UpdateAsync(order);

        var orderResponse = order.ToOrderResponse();
        await SendOkAsync(orderResponse, ct);
    }
}
