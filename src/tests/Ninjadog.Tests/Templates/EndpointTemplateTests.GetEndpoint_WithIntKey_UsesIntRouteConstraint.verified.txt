using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class GetOrderEndpoint(IOrderService orderService)
    : Endpoint<GetOrderRequest, OrderResponse>
{
    public override void Configure()
    {
        Get("/orders/{id:int}");
        AllowAnonymous();
    }

    public override async Task HandleAsync(GetOrderRequest req, CancellationToken ct)
    {
        var order = await orderService.GetAsync(req.OrderId);

        if (order is null)
        {
            await SendNotFoundAsync(ct);
            return;
        }

        var orderResponse = order.ToOrderResponse();
        await SendOkAsync(orderResponse, ct);
    }
}
