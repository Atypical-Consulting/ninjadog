using TestApp.Api.Contracts.Requests;
using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class UpdateTodoItemEndpoint(ITodoItemService todoItemService)
    : Endpoint<UpdateTodoItemRequest, TodoItemResponse>
{
    public override void Configure()
    {
        Put("/todo-items/{id:guid}");
        AllowAnonymous();
        Version(1);
    }

    public override async Task HandleAsync(UpdateTodoItemRequest req, CancellationToken ct)
    {
        var existingTodoItem = await todoItemService.GetAsync(req.Id);

        if (existingTodoItem is null)
        {
            await Send.NotFoundAsync(ct);
            return;
        }

        var todoItem = req.ToTodoItem();
        await todoItemService.UpdateAsync(todoItem);

        var todoItemResponse = todoItem.ToTodoItemResponse();
        await Send.OkAsync(todoItemResponse, ct);
    }
}
