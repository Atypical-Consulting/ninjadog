using TestApp.Api.Contracts.Requests;
using TestApp.Api.Services;
using FastEndpoints;
using Microsoft.AspNetCore.Authorization;

namespace TestApp.Api.Endpoints;

public partial class DeleteTodoItemEndpoint
    : Endpoint<DeleteTodoItemRequest>
{
    public ITodoItemService TodoItemService { get; private set; } = null!;

    public override void Configure()
    {
        Delete("/todo-items/{id:guid}");
        AllowAnonymous();
    }

    public override async Task HandleAsync(DeleteTodoItemRequest req, CancellationToken ct)
    {
        var deleted = await TodoItemService.DeleteAsync(req.Id);
        if (!deleted)
        {
            await SendNotFoundAsync(ct);
            return;
        }

        await SendNoContentAsync(ct);
    }
}
