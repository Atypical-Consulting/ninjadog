using FluentValidation;

namespace TestApp.Api.Validation;

public partial class UpdateContactRequestValidator : Validator<UpdateContactRequest>
{
    public UpdateContactRequestValidator()
    {
        RuleFor(x => x.Name)
            .NotEmpty().WithMessage("Name is required!")
            .MaximumLength(100)
            .MinimumLength(2);

        RuleFor(x => x.Email)
