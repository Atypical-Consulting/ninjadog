using TestApp.Api.Contracts.Responses;
using TestApp.Api.Mapping;
using TestApp.Api.Services;
using FastEndpoints;

namespace TestApp.Api.Endpoints;

public partial class GetAllOrdersEndpoint(IOrderService orderService)
    : EndpointWithoutRequest<GetAllOrdersResponse>
{
    public override void Configure()
    {
        Get("/orders");
        AllowAnonymous();
    }

    public override async Task HandleAsync(CancellationToken ct)
    {
        var page = int.TryParse(HttpContext.Request.Query["page"], out var p) && p > 0 ? p : 1;
        var pageSize = int.TryParse(HttpContext.Request.Query["pageSize"], out var ps) && ps > 0 ? ps : 10;

        var (orders, totalCount) = await orderService.GetAllAsync(page, pageSize);
        var ordersResponse = orders.ToOrdersResponse(page, pageSize, totalCount);
        await SendOkAsync(ordersResponse, ct);
    }
}
